function submitForm(event) { event.preventDefault(); const formData = new FormData(event.target); const formResponses = {}; formData.forEach((value, key) => { formResponses[key] = value; }); // Get current timestamp const timestamp = new Date().toISOString(); formResponses.timestamp = timestamp; // Fetch existing data from data.json fetch('data.json') .then(response => response.json()) .then(existingData => { if (!Array.isArray(existingData)) { existingData = []; } existingData.push(formResponses); // Update the data.json file with new array createOrUpdateDataJSON(existingData); }) .catch(error => console.error('Error fetching data.json:', error)); } function createOrUpdateDataJSON(data) { const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' }); const file = new File([blob], 'data.json', { type: 'application/json' }); const url = 'https://api.github.com/repos/lubeyond61-del/lubeyond61-del.github.io/contents/data.json'; const token = 'YOUR_GITHUB_TOKEN'; // Add your GitHub token fetch(url, { method: 'PUT', headers: { 'Authorization': `token ${token}`, 'Content-Type': 'application/json', }, body: JSON.stringify({ message: 'Update data.json with new response', content: btoa(blob), sha: 'YOUR_DATA_JSON_SHA' // Replace with the actual SHA }) }) .then(response => response.json()) .then(data => console.log('Data.json updated:', data)) .catch(error => console.error('Error updating data.json:', error)); }
